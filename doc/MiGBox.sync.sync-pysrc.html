 75      If the files given by C{src_path} and C{dst_path} are equal, nothing

 78      @param src: source file system abstraction. 
 79      @type src: L{MiGBox.FileSystem} 
 80      @param src_path: source path. 
 81      @type src_path: str 
 82      @param dst: destination file system abstraction. 
 83      @type dst: L{MiGBox.FileSystem} 
 84      @param dst_path: destination path. 
 85      @type dst_path: str 
 86      """ 
 87   
 88      try: 
 89          dst_mod, dst_bs = dst.cached_checksums(dst_path) 
 90          src_mod, src_bs = src.cached_checksums(src_path) 
 91          if dst_mod: 
 92              logging.info(_log['sync_conf'].format(src_path,dst_path)) 
 93          if set(src_bs) - set(dst_bs): 
 94              delta = src.delta(src_path, dst_bs) 
 95              dst.patch(dst_path, delta)

 97          else:
